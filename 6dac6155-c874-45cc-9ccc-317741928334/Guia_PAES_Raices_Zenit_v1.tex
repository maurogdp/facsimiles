\begin{tcolorbox}[importantbox, title={Instrucciones generales}]
\begin{enumerate}[label=\textbf{\arabic*.}, itemsep=4pt]
  \item Esta guía contiene \textbf{50 preguntas} de selección múltiple, con \textbf{5 alternativas} (A, B, C, D y E), de las cuales \textbf{una sola} es la respuesta correcta.
  \item Las preguntas \textbf{33} y \textbf{41} son de \textbf{Suficiencia de Datos}: determina si la información entregada en las condiciones (1) y/o (2) es suficiente para resolver el problema planteado.
  \item En los ítems de Suficiencia de Datos, las alternativas son siempre:
  \begin{itemize}[itemsep=2pt]
    \item[\textbf{A)}] La condición (1) por sí sola.
    \item[\textbf{B)}] La condición (2) por sí sola.
    \item[\textbf{C)}] Ambas condiciones juntas, (1) y (2).
    \item[\textbf{D)}] Cada condición por sí sola, (1) o (2).
    \item[\textbf{E)}] Se requiere información adicional.
  \end{itemize}
  \item Tiempo sugerido: \textbf{90 minutos}.
  \item Usa el espacio en blanco de cada página para tus cálculos.
  \item Está permitido el uso de calculadora científica, salvo que el enunciado indique lo contrario.
\end{enumerate}
\end{tcolorbox}
